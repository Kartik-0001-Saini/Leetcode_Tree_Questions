#include<bits/stdc++.h>
using namespace std;
class Node{
    public:
    int data;
    Node* left;
    Node* right;
    Node(int val){
        data = val;
        left = NULL;
        right = NULL;
    }
};

static int idx = -1;
Node* buildTree(vector<int> preOrder){
    idx++;
    if(preOrder[idx] == -1){
        return NULL;
    }
    
    Node* root = new Node(preOrder[idx]);
    root->left = buildTree(preOrder);
    root->right = buildTree(preOrder);
    return root;
}

// For pre-Order Traversal
void preOrderTra(Node* root){
    if(root == NULL){
        return;
    }
    
    cout<<root->data<<" ->  ";
    preOrderTra(root->left);
    preOrderTra(root->right);
}

// For In-Order Traversal
void inOrder(Node* root){
    if(root == NULL){
        return;
    }
    
    inOrder(root->left);
    cout<<root->data<<"  ->  ";
    inOrder(root->right);
}

// For Post-Order Traversal
void postOrder(Node* root){
    if(root == NULL){
        return;
    }
    
    postOrder(root->left);
    postOrder(root->right);
    cout<<root->data<<"  ->  ";
}

// For Level Order Traversal
void levelOrder(Node* root){
    queue<Node*> q;
    q.push(root);
    while(!q.empty()){
        Node* curr = q.front();
        q.pop();
        cout<<curr->data<<"  ->  ";
        if(curr->left != NULL){
            q.push(curr->left);
        }
        if(curr->right != NULL){
            q.push(curr->right);
        }
    }
    
    cout<<"NULL"<<endl;
}

int main(){
    vector<int> preOrder = {1,2,-1,-1,3,4,-1,-1,5,-1,-1};
    
    Node* root = buildTree(preOrder);
    cout<<root->left->data<<endl;
    
    preOrderTra(root);
    cout<<"NULL"<<endl;
    
    inOrder(root);
    cout<<"NULL"<<endl;
    
    postOrder(root);
    cout<<"NULL"<<endl;
    
    levelOrder(root);
    
    return 0;
}
